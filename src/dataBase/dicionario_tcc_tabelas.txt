  Campo fisico: log_id
  Tipo de dados: INT
  Tamanho do campo: -
  PK: Sim
  FK: Nao
  Restricoes: PRIMARY KEY; AUTO_INCREMENT; NOT NULL
  Observacoes: Identificador único do log.

- Campo logico: ID do usuario
  Campo fisico: usu_id
  Tipo de dados: INT
  Tamanho do campo: -
  PK: Nao
  FK: Sim
  Restricoes: NOT NULL; FK para Usuario(usu_id)
  Observacoes: Usuario que executou a acao.

- Campo logico: Acao
  Campo fisico: log_acao
  Tipo de dados: VARCHAR
  Tamanho do campo: 200
  PK: Nao
  FK: Nao
  Restricoes: NOT NULL
  Observacoes: Acao executada.

- Campo logico: Detalhe
  Campo fisico: log_detalhe
  Tipo de dados: VARCHAR
  Tamanho do campo: 255
  PK: Nao
  FK: Nao
  Restricoes: NOT NULL
  Observacoes: Detalhes da acao registrada.

- Campo logico: Data e hora
  Campo fisico: log_datahora
  Tipo de dados: DATETIME
  Tamanho do campo: -
  PK: Nao
  FK: Nao
  Restricoes: NOT NULL
  Observacoes: Data e hora do registro.
